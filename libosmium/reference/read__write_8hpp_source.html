118  inline bool reliable_read(const int fd, unsigned char* input_buffer, const size_t size) {
119  size_t offset = 0;
120  while (offset < size) {
121  ssize_t nread = ::read(fd, input_buffer + offset, size - offset);
122  if (nread < 0) {
123  throw std::system_error(errno, std::system_category(), "Read failed");
124  }
125  if (nread == 0) {
126  return false;
127  }
128  offset += nread;
129  }
130  return true;
131  }
